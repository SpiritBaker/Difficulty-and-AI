namespace = dev_events

country_event = {
	# adm_click
	id = dev_events.3
	title = dev_events.3.t
	desc = dev_events.3.d
	picture = INVENTION_eventPicture
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				ai = yes
				has_global_flag = ai_dev_click
			}
			AND = {
				NOT = { ai = yes }
				has_country_flag = dev_click
			}
		}
		
		OR = {
			had_country_flag = { flag = admClick	days = 31 }
			NOT = { has_country_flag = admClick }
		}
		
		power_tech = {
			power = adm
			value = 960
		}
	}
	
	immediate = {
		hidden_effect = {
			clr_country_flag = admClick
			set_country_flag = admClick
			
			set_base_target = {
				which = base_target
				base = adm
			}
			if = {
				limit = {
					OR = {
						is_migratory_tribe = yes
						check_variable = {
							which = baseCost
							value = 999
						}
					}
				}
				if = {
					limit = {
						NOT = { stability = 3 }
					}
					stab_cost = yes
					add_stability = 1
					log = "[This.GetName] raises stability**"
				}
				else = {
					adm_power_cost = 75
					add_inflation = -2
					log = "[This.GetName] reduces inflation**"
				}
			}
			else = {
				event_target:base_target = {
					add_power0 = {
						power = adm
					}
					add_next_institution_embracement_scaled = 5
					add_base_tax = 1
					add_devastation_reduction = yes
					log = dev_info.1
				}
				change_estate_land_share = {
					estate = all
					share = -0.2
				}
				change_variable = {
					which = admClicks
					value = 1
				}
				change_variable = {
					which = devsumClicks
					value = 1
				}
				change_variable = {
					which = autoDev
					value = 1
				}
			}
		}
	}
	
	option = {
		#hidden
	}
}

country_event = {
	#gold dev
	id = dev_events.6
	title = dev_events.6.t
	desc = dev_events.6.d
	picture = INVENTION_eventPicture
	hidden = yes
	
	mean_time_to_happen = {
		months = 2
	}
	
	trigger = {
		NOT = { has_global_flag = dev_click_disabled_flag }
		ai = yes
		is_at_war = no
		primitives = no
		is_lacking_institutions = no
		NOT = { is_migratory_tribe = yes }
		OR = {
			AND = {
				has_institution = feudalism 
				dip_power = 100
				NOT = { government = tribal }
				any_owned_province = {
					is_state_core = ROOT
					trade_goods = gold
					NOT = { base_production = 8 }
				}
			}
			AND = {
				has_institution = feudalism 
				dip_power = 900
				NOT = { government = tribal }
				any_owned_province = {
					is_state_core = ROOT
					trade_goods = gold
					NOT = { base_production = 11 }
				}
			}
			AND = {
				dip_power = 100
				any_owned_province = {
					is_state_core = ROOT
					trade_goods = gold
					NOT = { base_production = 6 }
				}
			}
		}
	}
	
	immediate = {
		random_owned_province = {
			limit = {
				OR = {
					AND = {
						ROOT = { 
							has_institution = feudalism 
							dip_power = 100
							NOT = { government = tribal }
						}
						is_state_core = ROOT
						trade_goods = gold
						NOT = { base_production = 8 }
					}
					AND = {
						ROOT = { 
							has_institution = feudalism  
							dip_power = 900
							NOT = { government = tribal }
						}
						is_state_core = ROOT
						trade_goods = gold
						NOT = { base_production = 11 }
					}
					AND = {
						ROOT = {
							dip_power = 100
						}
						is_state_core = ROOT
						trade_goods = gold
						NOT = { base_production = 6 }
					}
				}
			}
			set_base_value0 = {
				which = baseCost
			}
			
			export_to_variable = { which = bTax 	value = base_tax }
			export_to_variable = { which = bPro 	value = base_production }
			export_to_variable = { which = bMan 	value = base_manpower }
			
			set_variable 	= { which = notPro 	which = bMan }
			change_variable = { which = notPro 	which = bTax }
			
			export_to_variable = { which = admPower 	value = trigger_value:adm_power 	who = PREV }
			export_to_variable = { which = milPower 	value = trigger_value:mil_power 	who = PREV }
			
			add_next_institution_embracement_scaled = 5
			
			if = {
				limit = {
					check_variable = {
						which = notPro
						which = bPro
					}
				}
				add_power0 = {
					power = dip
				}
				add_base_production = 1
				log = dev_info.2
				owner = {
					change_variable = {
						which = dipClicks
						value = 1
					}
					change_variable = {
						which = devsumClicks
						value = 1
					}
				}
			}
			else = {
				if = {
					limit = {
						check_variable = {
							which = admPower
							which = milPower
						}
					}
					add_power0 = {
						power = adm
					}
					add_base_tax = 1
					log = dev_info.1
					owner = {
						change_variable = {
							which = admClicks
							value = 1
						}
						change_variable = {
							which = devsumClicks
							value = 1
						}
					}
				}
				else = {
					add_power0 = {
						power = mil
					}
					add_base_manpower = 1
					log = dev_info.3
					owner = {
						change_variable = {
							which = milClicks
							value = 1
						}
						change_variable = {
							which = devsumClicks
							value = 1
						}
					}
				}
			}
			
			add_devastation_reduction = yes
			owner = {
				change_estate_land_share = {
					estate = all
					share = -0.2
				}
			}
		}
	}
	
	option = {
		#hidden
	}
}

province_event = {
	# dev cap
	id = dev_events.7
	title = dev_events.7.t
	desc = dev_events.7.d
	picture = CITY_VIEW_eventPicture
	goto = ROOT
	
	mean_time_to_happen = {
		months = 2
	}
	
	trigger = {
		always = no 
		OR = {
			AND = {
				development = 50
				NOT = { development = 60 }
				NOT = { has_province_modifier = regiopolis_city }
			}
			AND = {
				development = 60
				NOT = { development = 70 }
				NOT = { has_province_modifier = metropolis_city }
			}
			AND = {
				development = 70
				NOT = { has_province_modifier = world_city }
			}
		}
	}
	
	option = {
		name = dev_events.7.a
		
		if = {
			limit = {
				development = 50
				NOT = { development = 60 }
				NOT = { has_province_modifier = regiopolis_city }
			}
			remove_province_modifier = metropolis_city
			remove_province_modifier = world_city
			add_province_modifier = {
				name = regiopolis_city
				duration = -1
			}
		}
		else_if = {
			limit = {
				development = 60
				NOT = { development = 70 }
				NOT = { has_province_modifier = metropolis_city }
			}
			remove_province_modifier = regiopolis_city
			remove_province_modifier = world_city
			add_province_modifier = {
				name = metropolis_city
				duration = -1
			}
		}
		else_if = {
			limit = {
				development = 70
				NOT = { has_province_modifier = world_city }
			}
			remove_province_modifier = regiopolis_city
			remove_province_modifier = metropolis_city
			add_province_modifier = {
				name = world_city
				duration = -1
			}
		}
	}
}
